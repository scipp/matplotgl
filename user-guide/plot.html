import matplotgl.pyplot as plt
import numpy as np

[2]:

fig, ax = plt.subplots()

x = np.arange(50.)
y = np.sin(0.2 * x)

ax.plot(x, y, lw=2)
ax.plot(x, 0.5 * y, lw=2)
ax.set_xlabel('Time (seconds)')
ax.set_ylabel('Amplitude (cm)')
